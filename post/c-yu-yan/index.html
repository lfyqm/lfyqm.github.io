题目描述:从文件a.txt中读取三个整数，然后把这三个整数保存到b.txt中，两整数之间一个空格。
#include<stdio.h>
void solve(){
/********** Begin *********/
int one,two,three;
FILE *a,*b;
a=fopen("a.txt","r");
b=fopen("b.txt","w");
int n=0;
while(n!=1)
{
fscanf(a,"%d%d%d",&one,&two,&three);
fprintf(b,"%d %d %d",one,two,three);
n++;
}
fclose(a);
fclose(b);
}
